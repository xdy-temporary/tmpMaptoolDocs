   {{MacroFunction
   |name=max
   |description=
   Returns the largest number of all the numbers that have been passed in.

   |usage=
   <source lang="mtmacro" line>
   [h: largest = max(num, num, ...)]
   </source>

   |examples=
   <source lang="mtmacro" line>
       [r: max(2, 30, 4, 7)]
       [r: max(-2, -40, -100)]
   </source>
   Returns 
       30
       -2
   }}
