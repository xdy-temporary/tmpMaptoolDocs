   {{MacroFunction
   |name=min
   |description=
   Returns the smallest of the numbers that are passed in.
       
   |usage=
   <source lang="mtmacro" line>
   [h: smallest = min(num, num, ...)]
   </source>

   |example=
   <source lang="mtmacro" line>
       [r: min(2, 4, 5, 1)]   
       [r: min(-1, -5, 100)]
   </source>
   Returns 
       1
       -5
   }}
