   {{MacroFunction
   |name=squareroot
   |description=
   Returns the [[wp:Square_root|square root]] of a number.

   |usage=
   <source lang="mtmacro" line>
   [h: num = squareroot(num)]
   [h: num = sqrt(num)]
   </source>

   |example=
   <source lang="mtmacro" line>
   [r: squareroot(9)]
   [r: squareroot(81)]
   </source>
   Returns 
       3
       9
   }}
